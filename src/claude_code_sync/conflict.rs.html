9/// Represents a conflict between local and remote versions of the same conversation session.
10///
11/// A `Conflict` is detected when both local and remote filesystems contain a conversation
12/// with the same session ID but different content (as determined by content hashes). This
13/// typically occurs when the same conversation has been modified on different machines or
14/// when changes haven't been synchronized properly.
15///
16/// The conflict contains metadata about both versions to help users make informed decisions
17/// about how to resolve the discrepancy.
18///
19/// # Examples
20///
21/// ```ignore
22/// use claude_code_sync::conflict::Conflict;
23/// use claude_code_sync::parser::ConversationSession;
24///
25/// let conflict = Conflict::new(local, remote);
26///
27/// if conflict.is_real_conflict() {
28///     println!("{}", conflict.description());
29/// }
30/// ```
31#[derive(Debug, Clone, Serialize, Deserialize)]
32pub struct Conflict {
33    /// The unique identifier for the conversation session that has conflicted.
34    ///
35    /// This ID is shared between both the local and remote versions, as they represent
36    /// different states of the same conversation.
37    pub session_id: String,
38
39    /// The file path to the local version of the conversation.
40    ///
41    /// This points to the conversation file in the local filesystem, typically in the
42    /// user's local conversation storage directory.
43    pub local_file: PathBuf,
44
45    /// The file path to the remote version of the conversation.
46    ///
47    /// This points to the conversation file from the remote source (e.g., synced from
48    /// another machine or cloud storage).
49    pub remote_file: PathBuf,
50
51    /// The timestamp of the most recent message in the local version.
52    ///
53    /// This is `None` if the local conversation has no messages with timestamps.
54    /// The timestamp helps users understand which version is more recent.
55    pub local_timestamp: Option<String>,
56
57    /// The timestamp of the most recent message in the remote version.
58    ///
59    /// This is `None` if the remote conversation has no messages with timestamps.
60    /// The timestamp helps users understand which version is more recent.
61    pub remote_timestamp: Option<String>,
62
63    /// The total number of messages in the local version of the conversation.
64    ///
65    /// This count includes all conversation entries (user messages, assistant responses, etc.)
66    /// and helps users compare the relative completeness of each version.
67    pub local_message_count: usize,
68
69    /// The total number of messages in the remote version of the conversation.
70    ///
71    /// This count includes all conversation entries (user messages, assistant responses, etc.)
72    /// and helps users compare the relative completeness of each version.
73    pub remote_message_count: usize,
74
75    /// A hash of the local conversation's content.
76    ///
77    /// This hash is used to detect whether the local and remote versions are truly different.
78    /// If the hashes match, the conversations are identical despite any metadata differences.
79    pub local_hash: String,
80
81    /// A hash of the remote conversation's content.
82    ///
83    /// This hash is used to detect whether the local and remote versions are truly different.
84    /// If the hashes match, the conversations are identical despite any metadata differences.
85    pub remote_hash: String,
86
87    /// The current resolution status of the conflict.
88    ///
89    /// Initially set to `ConflictResolution::Pending` when a conflict is detected.
90    /// Updated to one of the other variants once the user or system decides how to
91    /// resolve the conflict.
92    pub resolution: ConflictResolution,
93}

95/// Represents the resolution strategy for a conversation conflict.
96///
97/// When a conflict is detected between local and remote versions of the same conversation,
98/// the user or system must choose how to resolve it. This enum captures the different
99/// resolution strategies available.
100///
101/// # Resolution Strategies
102///
103/// - **SmartMerge**: Intelligently combines both versions by merging non-conflicting changes
104/// - **KeepBoth**: Preserves both versions by renaming the remote file to avoid overwriting
105/// - **KeepLocal**: Discards the remote version and keeps only the local version
106/// - **KeepRemote**: Discards the local version and keeps only the remote version
107/// - **Pending**: No resolution has been chosen yet (default state for new conflicts)
108#[derive(Debug, Clone, Serialize, Deserialize)]
109pub enum ConflictResolution {
110    /// Intelligently merge both versions into a single conversation.
111    ///
112    /// This strategy attempts to combine messages from both local and remote versions
113    /// by analyzing message UUIDs, parent relationships, and timestamps. It can handle:
114    /// - Non-overlapping messages (simple merge)
115    /// - Edited messages (resolved by timestamp)
116    /// - Conversation branches (all branches preserved)
117    /// - Entries without UUIDs (merged by timestamp)
118    ///
119    /// If smart merge fails (e.g., due to corrupted data or circular references),
120    /// the system will fall back to another resolution strategy.
121    ///
122    /// # Fields
123    ///
124    /// * `merged_entries` - The result of merging both conversations
125    /// * `stats` - Statistics about the merge operation
126    SmartMerge {
127        /// The merged conversation entries
128        merged_entries: Vec<crate::parser::ConversationEntry>,
129        /// Statistics about the merge operation
130        stats: merge::MergeStats,
131    },
132
133    /// Keep both versions by renaming the remote file with a conflict suffix.
134    ///
135    /// This strategy preserves both the local and remote versions of the conversation.
136    /// The local file retains its original name and location, while the remote file
137    /// is renamed to include a conflict marker (typically a timestamp-based suffix)
138    /// to prevent overwriting the local version.
139    ///
140    /// # Fields
141    ///
142    /// * `renamed_remote_file` - The new path where the remote file will be saved.
143    ///   This path includes a conflict suffix to distinguish it from the local version.
144    ///
145    /// # Example
146    ///
147    /// If the local file is `conversation.jsonl` and a conflict is detected,
148    /// the remote version might be saved as `conversation-conflict-20250122-143000.jsonl`.
149    KeepBoth {
150        /// The destination path for the renamed remote file, including the conflict suffix.
151        renamed_remote_file: PathBuf,
152    },
153
154    /// Keep only the local version and discard the remote version.
155    ///
156    /// This strategy assumes the local version is correct and the remote version
157    /// should be ignored. The local file remains unchanged, and the remote version
158    /// is not saved to disk.
159    KeepLocal,
160
161    /// Keep only the remote version and discard the local version.
162    ///
163    /// This strategy assumes the remote version is correct and should replace the
164    /// local version. The local file will be overwritten with the remote content.
165    KeepRemote,
166
167    /// The conflict has not yet been resolved.
168    ///
169    /// This is the default state for newly detected conflicts. The user must choose
170    /// one of the other resolution strategies before the conflict can be resolved.
171    Pending,
172}

174impl Conflict {
175    /// Creates a new `Conflict` by comparing local and remote conversation sessions.
176    ///
177    /// This function constructs a conflict record from two versions of the same conversation
178    /// session (identified by matching session IDs). It extracts and stores relevant metadata
179    /// from both versions, including timestamps, message counts, and content hashes.
180    ///
181    /// The conflict is initialized with a `Pending` resolution status, indicating that no
182    /// resolution strategy has been chosen yet.
183    ///
184    /// # Arguments
185    ///
186    /// * `local` - A reference to the local version of the conversation session
187    /// * `remote` - A reference to the remote version of the conversation session
188    ///
189    /// # Returns
190    ///
191    /// A new `Conflict` instance containing metadata from both conversation versions,
192    /// with the resolution status set to `Pending`.
193    ///
194    /// # Examples
195    ///
196    /// ```ignore
197    /// use claude_code_sync::conflict::Conflict;
198    /// use claude_code_sync::parser::ConversationSession;
199    ///
200    /// let conflict = Conflict::new(local_session, remote_session);
201    ///
202    /// println!("Conflict in session: {}", conflict.session_id);
203    /// println!("Local messages: {}", conflict.local_message_count);
204    /// println!("Remote messages: {}", conflict.remote_message_count);
205    /// ```
206    pub fn new(local: &ConversationSession, remote: &ConversationSession) -> Self {
207        Conflict {
208            session_id: local.session_id.clone(),
209            local_file: PathBuf::from(&local.file_path),
210            remote_file: PathBuf::from(&remote.file_path),
211            local_timestamp: local.latest_timestamp(),
212            remote_timestamp: remote.latest_timestamp(),
213            local_message_count: local.message_count(),
214            remote_message_count: remote.message_count(),
215            local_hash: local.content_hash(),
216            remote_hash: remote.content_hash(),
217            resolution: ConflictResolution::Pending,
218        }
219    }
220
221    /// Attempts to resolve the conflict using smart merge
222    ///
223    /// This method tries to intelligently combine local and remote versions
224    /// by analyzing message UUIDs, timestamps, and parent relationships.
225    ///
226    /// # Arguments
227    ///
228    /// * `local_session` - The local conversation session
229    /// * `remote_session` - The remote conversation session
230    ///
231    /// # Returns
232    ///
233    /// Returns `Ok(())` if the smart merge succeeds, or an error if it fails.
234    /// On success, the conflict resolution is set to `SmartMerge` with the merged entries.
235    pub fn try_smart_merge(
236        &mut self,
237        local_session: &ConversationSession,
238        remote_session: &ConversationSession,
239    ) -> Result<()> {
240        let merge_result = merge::merge_conversations(local_session, remote_session)?;
241
242        self.resolution = ConflictResolution::SmartMerge {
243            merged_entries: merge_result.merged_entries,
244            stats: merge_result.stats,
245        };
246
247        Ok(())
248    }
249
250    /// Resolve the conflict by keeping both versions
251    pub fn resolve_keep_both(&mut self, conflict_suffix: &str) -> Result<PathBuf> {
252        let remote_file_name = self
253            .remote_file
254            .file_stem()
255            .and_then(|s| s.to_str())
256            .unwrap_or("unknown");
257
258        let remote_file_ext = self
259            .remote_file
260            .extension()
261            .and_then(|s| s.to_str())
262            .unwrap_or("jsonl");
263
264        let parent = self.remote_file.parent().unwrap_or_else(|| Path::new("."));
265
266        let new_name = format!(
267            "{remote_file_name}-{conflict_suffix}.{remote_file_ext}"
268        );
269        let renamed_path = parent.join(new_name);
270
271        self.resolution = ConflictResolution::KeepBoth {
272            renamed_remote_file: renamed_path.clone(),
273        };
274
275        Ok(renamed_path)
276    }

278    /// Get a human-readable description of the conflict
279    pub fn description(&self) -> String {
280        format!(
281            "Session {} has diverged:\n  Local: {} messages, last update: {}\n  Remote: {} messages, last update: {}",
282            self.session_id,
283            self.local_message_count,
284            self.local_timestamp.as_deref().unwrap_or("unknown"),
285            self.remote_message_count,
286            self.remote_timestamp.as_deref().unwrap_or("unknown")
287        )
288    }

290    /// Determine if this is a real conflict (different content)
291    pub fn is_real_conflict(&self) -> bool {
292        self.local_hash != self.remote_hash
293    }

303    ///
304    /// The conflict detector is used to identify and manage conflicts between local and remote
305    /// conversation sessions. It maintains a list of detected conflicts and provides methods
306    /// to resolve them according to different strategies.
307    ///
308    /// The detector starts with no conflicts; conflicts are added by calling the [`detect`]
309    /// method with local and remote conversation sessions.
310    ///
311    /// [`detect`]: ConflictDetector::detect
312    ///
313    /// # Returns
314    ///
315    /// A new `ConflictDetector` instance with an empty internal conflict list, ready to
316    /// detect and manage conflicts between conversation sessions.
317    ///
318    /// # Examples
319    ///
320    /// ```

322    /// # use claude_code_sync::parser::ConversationSession;

324    /// let mut detector = ConflictDetector::new();
325    ///
326    /// // Detect conflicts between local and remote sessions
327    /// detector.detect(&local_sessions, &remote_sessions);
328    ///
329    /// if detector.has_conflicts() {
330    ///     println!("Found {} conflicts", detector.conflict_count());
331    /// }
332    /// # }
333    /// ```
334    ///
335    /// # See Also
336    ///
337    /// * [`detect`] - Method to scan for conflicts between local and remote sessions
338    /// * [`has_conflicts`] - Check if any conflicts have been detected
339    /// * [`conflict_count`] - Get the number of detected conflicts
340    ///
341    /// [`detect`]: ConflictDetector::detect
342    /// [`has_conflicts`]: ConflictDetector::has_conflicts
343    /// [`conflict_count`]: ConflictDetector::conflict_count

345        ConflictDetector {
346            conflicts: Vec::new(),
347        }

350    /// Compare local and remote sessions and detect conflicts
351    ///
352    /// Only reports TRUE conflicts where both sides have diverged.
353    /// Simple extensions (one side has more messages) are NOT conflicts.
354    pub fn detect(
355        &mut self,
356        local_sessions: &[ConversationSession],
357        remote_sessions: &[ConversationSession],
358    ) {
359        // Build a map of session_id -> local session
360        let local_map: std::collections::HashMap<_, _> = local_sessions
361            .iter()
362            .map(|s| (s.session_id.clone(), s))
363            .collect();
364
365        // Check each remote session against local
366        for remote in remote_sessions {
367            if let Some(local) = local_map.get(&remote.session_id) {
368                // Session exists in both - analyze relationship
369                let relationship = analyze_session_relationship(local, remote);
370
371                match relationship {
372                    SessionRelationship::Identical => {
373                        // No action needed - sessions are the same
374                    }
375                    SessionRelationship::LocalIsPrefix => {
376                        // Remote has more messages - NOT a conflict
377                        // This will be handled as a normal "Modified" copy in pull
378                        log::debug!(
379                            "Session {} is extended in remote ({} -> {} entries)",
380                            local.session_id,
381                            local.entries.len(),
382                            remote.entries.len()
383                        );
384                    }
385                    SessionRelationship::RemoteIsPrefix => {
386                        // Local has more messages - NOT a conflict
387                        // Keep local, no action needed during pull
388                        log::debug!(
389                            "Session {} is extended locally ({} -> {} entries), keeping local",
390                            local.session_id,
391                            remote.entries.len(),
392                            local.entries.len()
393                        );
394                    }
395                    SessionRelationship::Diverged => {
396                        // TRUE conflict - both have unique entries
397                        let conflict = Conflict::new(local, remote);
398                        self.conflicts.push(conflict);
399                        log::info!(
400                            "True conflict detected in session {} (local: {}, remote: {} entries)",
401                            local.session_id,
402                            local.entries.len(),
403                            remote.entries.len()
404                        );
405                    }
406                }
407            }
408        }
409    }

411    /// Resolve all conflicts using the "keep both" strategy
412    #[allow(dead_code)]
413    pub fn resolve_all_keep_both(&mut self) -> Result<Vec<(PathBuf, PathBuf)>> {
414        let mut renames = Vec::new();
415
416        for conflict in &mut self.conflicts {
417            let timestamp = chrono::Utc::now().format("%Y%m%d-%H%M%S");
418            let conflict_suffix = format!("conflict-{timestamp}");
419
420            let renamed_path = conflict.resolve_keep_both(&conflict_suffix)?;
421            renames.push((conflict.remote_file.clone(), renamed_path));
422        }
423
424        Ok(renames)
425    }

454/// Relationship between two sessions with the same ID
455#[derive(Debug, Clone, PartialEq, Eq)]
456pub enum SessionRelationship {
457    /// Sessions are identical (same content hash)
458    Identical,
459    /// Local is a prefix of remote (remote has more messages, all local messages exist in remote)
460    LocalIsPrefix,
461    /// Remote is a prefix of local (local has more messages, all remote messages exist in local)
462    RemoteIsPrefix,
463    /// True divergence - both have unique messages not in the other (actual conflict)
464    Diverged,
465}
466
467/// Analyzes the relationship between two sessions to determine if they truly conflict
468/// or if one is simply an extension of the other.
469///
470/// This is crucial for avoiding false "conflicts" when:
471/// - Machine A has session with 40 messages
472/// - Machine B has the same session with 50 messages (continued the conversation)
473/// - This is NOT a conflict - Machine B just has more messages
474///
475/// True conflicts only occur when BOTH sides have added different messages.
476pub fn analyze_session_relationship(
477    local: &ConversationSession,
478    remote: &ConversationSession,
479) -> SessionRelationship {
480    // Fast path: identical hashes
481    if local.content_hash() == remote.content_hash() {
482        return SessionRelationship::Identical;
483    }
484
485    // Build sets of UUIDs from each session
486    let local_uuids: HashSet<String> = local
487        .entries
488        .iter()
489        .filter_map(|e| e.uuid.clone())
490        .collect();
491
492    let remote_uuids: HashSet<String> = remote
493        .entries
494        .iter()
495        .filter_map(|e| e.uuid.clone())
496        .collect();
497
498    // Check for entries unique to each side
499    let local_only: HashSet<_> = local_uuids.difference(&remote_uuids).collect();
500    let remote_only: HashSet<_> = remote_uuids.difference(&local_uuids).collect();
501
502    // If local has no unique entries, local is a prefix of remote
503    if local_only.is_empty() && !remote_only.is_empty() {
504        // Verify common entries are identical
505        if verify_common_entries_identical(local, remote) {
506            return SessionRelationship::LocalIsPrefix;
507        }
508    }
509
510    // If remote has no unique entries, remote is a prefix of local
511    if remote_only.is_empty() && !local_only.is_empty() {
512        // Verify common entries are identical
513        if verify_common_entries_identical(local, remote) {
514            return SessionRelationship::RemoteIsPrefix;
515        }
516    }
517
518    // Both have unique entries - true divergence
519    SessionRelationship::Diverged
520}
521
522/// Verifies that entries with the same UUID have identical content
523fn verify_common_entries_identical(
524    local: &ConversationSession,
525    remote: &ConversationSession,
526) -> bool {
527    use std::collections::HashMap;
528
529    // Build map of UUID -> serialized entry for local
530    let local_map: HashMap<String, String> = local
531        .entries
532        .iter()
533        .filter_map(|e| {
534            e.uuid.as_ref().and_then(|uuid| {
535                serde_json::to_string(e).ok().map(|json| (uuid.clone(), json))
536            })
537        })
538        .collect();
539
540    // Check each remote entry with a UUID
541    for entry in &remote.entries {
542        if let Some(uuid) = &entry.uuid {
543            if let Some(local_json) = local_map.get(uuid) {
544                // This UUID exists in both - check if content is identical
545                if let Ok(remote_json) = serde_json::to_string(entry) {
546                    if &remote_json != local_json {
547                        // Same UUID but different content - entries were modified
548                        return false;
549                    }
550                }
551            }
552        }
553    }
554
555    true
556}
