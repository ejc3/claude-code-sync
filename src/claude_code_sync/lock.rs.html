1//! File-based locking to prevent concurrent sync operations.
2//!
3//! Uses `flock` (via fs2) to ensure only one sync runs at a time.
4
5use anyhow::{Context, Result};
6use fs2::FileExt;
7use std::fs::{File, OpenOptions};
8use std::path::PathBuf;
9
10use crate::config::ConfigManager;
11
12/// A guard that holds an exclusive lock on the sync lock file.
13/// The lock is released when this guard is dropped.
14pub struct SyncLock {
15    _file: File,
16    path: PathBuf,
17}
18
19impl SyncLock {
20    /// Attempt to acquire an exclusive lock for sync operations.
21    ///
22    /// Returns `Ok(SyncLock)` if the lock was acquired, or an error if
23    /// another sync is already running.
24    pub fn acquire() -> Result<Self> {
25        let lock_path = Self::lock_path()?;
26
27        // Ensure parent directory exists
28        if let Some(parent) = lock_path.parent() {
29            std::fs::create_dir_all(parent)
30                .with_context(|| format!("Failed to create lock directory: {}", parent.display()))?;
31        }
32
33        let file = OpenOptions::new()
34            .create(true)
35            .write(true)
36            .truncate(true)
37            .open(&lock_path)
38            .with_context(|| format!("Failed to open lock file: {}", lock_path.display()))?;
39
40        // Try to acquire exclusive lock (non-blocking)
41        match file.try_lock_exclusive() {
42            Ok(()) => {
43                log::debug!("Acquired sync lock: {}", lock_path.display());
44                Ok(Self {
45                    _file: file,
46                    path: lock_path,
47                })
48            }
49            Err(e) => {
50                Err(anyhow::anyhow!(
51                    "Another sync operation is already running. \
52                     If you're sure no other sync is running, delete the lock file: {}\n\
53                     Original error: {}",
54                    lock_path.display(),
55                    e
56                ))
57            }
58        }
59    }
60
61    fn lock_path() -> Result<PathBuf> {
62        let config_dir = ConfigManager::ensure_config_dir()?;
63        Ok(config_dir.join("sync.lock"))
64    }
65}
66
67impl Drop for SyncLock {
68    fn drop(&mut self) {
69        log::debug!("Releasing sync lock: {}", self.path.display());
70        // File lock is automatically released when the file is closed
71    }
72}

74#[cfg(test)]
75mod tests {
76    use super::*;
77    use serial_test::file_serial;
78    use std::env;
79    use tempfile::TempDir;
80
81    #[test]
82    #[file_serial]
83    fn test_lock_acquire_and_release() {
84        let temp_dir = TempDir::new().unwrap();
85        let original_home = env::var("HOME").ok();
86        env::set_var("HOME", temp_dir.path());
87
88        // First lock should succeed
89        let lock1 = SyncLock::acquire().unwrap();
90
91        // Second lock should fail
92        let lock2_result = SyncLock::acquire();
93        match lock2_result {
94            Err(e) => assert!(e.to_string().contains("already running")),
95            Ok(_) => panic!("Expected lock acquisition to fail"),
96        }
97
98        // Drop first lock
99        drop(lock1);
100
101        // Now we can acquire again
102        let _lock3 = SyncLock::acquire().unwrap();
103
104        // Restore HOME
105        if let Some(home) = original_home {
106            env::set_var("HOME", home);
107        } else {
108            env::remove_var("HOME");
109        }
110    }
