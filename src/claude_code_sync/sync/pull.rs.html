22/// Generate a unique temp branch name with timestamp
23fn generate_temp_branch_name() -> String {
24    let timestamp = chrono::Utc::now().format("%Y%m%d-%H%M%S");
25    format!("sync-local-{}", timestamp)
26}

28/// Pull and merge history from sync repository
29///
30/// Safe workflow:
31/// 1. Create temp branch from current state
32/// 2. Copy local .claude sessions to sync repo and commit to temp branch
33/// 3. Push temp branch to remote (preserves local work - SAFETY NET)
34/// 4. Checkout main/master and pull from remote
35/// 5. Merge temp branch into main (smart conflict resolution)
36/// 6. Copy merged result to .claude
37/// 7. Delete temp branch (local + remote)
38pub fn pull_history(
39    fetch_remote: bool,
40    branch: Option<&str>,
41    interactive: bool,
42    verbosity: crate::VerbosityLevel,
43) -> Result<()> {
44    use crate::VerbosityLevel;
45
46    // Acquire exclusive lock to prevent concurrent sync operations
47    let _lock = SyncLock::acquire()?;
48
49    if verbosity != VerbosityLevel::Quiet {
50        println!("{}", "Pulling Claude Code history...".cyan().bold());
51    }
52
53    let state = SyncState::load()?;
54    let repo = scm::open(&state.sync_repo_path)?;
55    let filter = FilterConfig::load()?;
56    let claude_dir = claude_projects_dir()?;
57
58    // Clean up old temp branches that have exceeded retention period
59    cleanup_old_temp_branches(
60        repo.as_ref(),
61        fetch_remote && state.has_remote,
62        filter.temp_branch_retention_hours,
63        verbosity,
64    )?;
65
66    // Get the main branch name
67    let main_branch = branch
68        .map(|s| s.to_string())
69        .or_else(|| repo.current_branch().ok())
70        .unwrap_or_else(|| "main".to_string());
71
72    // ============================================================================
73    // STEP 1: Create temp branch and save local state
74    // ============================================================================
75    let temp_branch = generate_temp_branch_name();
76
77    if verbosity != VerbosityLevel::Quiet {
78        println!("  {} temp branch '{}'...", "Creating".cyan(), temp_branch);
79    }
80
81    // Create the temp branch from current HEAD
82    repo.create_branch(&temp_branch)
83        .context("Failed to create temp branch")?;
84    repo.checkout(&temp_branch)
85        .context("Failed to checkout temp branch")?;
86
87    // ============================================================================
88    // STEP 2: Copy local .claude sessions to sync repo on temp branch
89    // ============================================================================
90    if verbosity != VerbosityLevel::Quiet {
91        println!("  {} local sessions to temp branch...", "Saving".cyan());
92    }
93
94    let local_sessions = discover_sessions(&claude_dir, &filter)?;
95    let projects_dir = state.sync_repo_path.join(&filter.sync_subdirectory);
96    std::fs::create_dir_all(&projects_dir)?;
97
98    let mut local_session_count = 0;
99    for session in &local_sessions {
100        let relative_path = Path::new(&session.file_path)
101            .strip_prefix(&claude_dir)
102            .unwrap_or(Path::new(&session.file_path));
103        let dest_path = projects_dir.join(relative_path);
104        session.write_to_file(&dest_path)?;
105        local_session_count += 1;
106    }
107
108    // Also copy history.jsonl to sync repo (session index for --resume picker)
109    let claude_base_dir = claude_dir.parent().unwrap_or(&claude_dir);
110    let local_history = claude_base_dir.join("history.jsonl");
111    let sync_history = state.sync_repo_path.join("history.jsonl");
112    if local_history.exists() {
113        // Merge local history into sync repo history (preserving remote entries)
114        let (total, added) = super::history_merge::merge_history_files(
115            &local_history,
116            &sync_history,
117            super::history_merge::MergePriority::TargetFirst,
118        )?;
119        log::debug!("Saved history.jsonl to sync repo: {} total, {} added", total, added);
120    }
121
122    // Commit local state to temp branch
123    repo.stage_all()?;
124    if repo.has_changes()? {
125        let commit_msg = format!(
126            "Save local state before pull ({})",
127            chrono::Utc::now().format("%Y-%m-%d %H:%M:%S UTC")
128        );
129        repo.commit(&commit_msg)?;
130
131        if verbosity != VerbosityLevel::Quiet {
132            println!(
133                "  {} Saved {} local sessions to temp branch",
134                "✓".green(),
135                local_session_count
136            );
137        }
138    } else if verbosity != VerbosityLevel::Quiet {
139        println!("  {} No local changes to save", "✓".green());
140    }
141
142    // ============================================================================
143    // STEP 3: Push temp branch to remote (SAFETY NET - never lose work)
144    // ============================================================================
145    if fetch_remote && state.has_remote {
146        if verbosity != VerbosityLevel::Quiet {
147            println!("  {} temp branch to remote...", "Pushing".cyan());
148        }
149
150        match repo.push("origin", &temp_branch) {
151            Ok(_) => {
152                if verbosity != VerbosityLevel::Quiet {
153                    println!("  {} Pushed temp branch to origin/{}", "✓".green(), temp_branch);
154                }
155            }
156            Err(e) => {
157                log::warn!("Failed to push temp branch: {}", e);
158                log::info!("Continuing - local temp branch still preserves your work");
159                if verbosity != VerbosityLevel::Quiet {
160                    println!(
161                        "  {} Could not push temp branch: {}",
162                        "!".yellow().bold(),
163                        e
164                    );
165                    println!(
166                        "  {} Local temp branch {} still preserves your work",
167                        "ℹ".cyan(),
168                        temp_branch
169                    );
170                }
171            }
172        }
173    }
174
175    // ============================================================================
176    // STEP 4: Checkout main and pull from remote
177    // ============================================================================
178    if verbosity != VerbosityLevel::Quiet {
179        println!("  {} to main branch...", "Switching".cyan());
180    }
181
182    repo.checkout(&main_branch)
183        .context("Failed to checkout main branch")?;
184
185    if fetch_remote && state.has_remote {
186        if verbosity != VerbosityLevel::Quiet {
187            println!("  {} from remote...", "Pulling".cyan());
188        }
189
190        let mut fetch_failed = false;
191        let mut pull_failed = false;
192
193        // First fetch to see what's on remote
194        match repo.fetch("origin") {
195            Ok(_) => {
196                if verbosity != VerbosityLevel::Quiet {
197                    println!("  {} Fetched from origin", "✓".green());
198                }
199            }
200            Err(e) => {
201                log::warn!("Failed to fetch: {}", e);
202                fetch_failed = true;
203                if verbosity != VerbosityLevel::Quiet {
204                    println!(
205                        "  {} Failed to fetch from origin: {}",
206                        "!".yellow().bold(),
207                        e
208                    );
209                }
210            }
211        }
212
213        // Now pull (which will fast-forward if possible)
214        match repo.pull("origin", &main_branch) {
215            Ok(_) => {
216                if verbosity != VerbosityLevel::Quiet {
217                    println!("  {} Pulled origin/{}", "✓".green(), main_branch);
218                }
219            }
220            Err(e) => {
221                log::warn!("Failed to pull: {}", e);
222                log::info!("Continuing with local state...");
223                pull_failed = true;
224                if verbosity != VerbosityLevel::Quiet {
225                    println!(
226                        "  {} Failed to pull from origin/{}: {}",
227                        "!".yellow().bold(),
228                        main_branch,
229                        e
230                    );
231                }
232            }
233        }
234
235        // Inform user if network operations failed
236        if (fetch_failed || pull_failed) && verbosity != VerbosityLevel::Quiet {
237            println!(
238                "  {} Continuing with local state (remote changes may not be included)",
239                "ℹ".cyan()
240            );
241        }
242    }
243
244    // ============================================================================
245    // STEP 5: Merge temp branch into main (smart merge)
246    // ============================================================================
247    if verbosity != VerbosityLevel::Quiet {
248        println!("  {} temp branch into main...", "Merging".cyan());
249    }
250
251    // Discover sessions from both branches
252    // - main branch now has remote changes
253    // - temp branch has our local changes
254    let remote_sessions = discover_sessions(&projects_dir, &filter)?;
255
256    // We need to get the local sessions from the temp branch
257    // Switch to temp branch, read sessions, switch back
258    repo.checkout(&temp_branch)?;
259    let temp_branch_sessions = discover_sessions(&projects_dir, &filter)?;
260    repo.checkout(&main_branch)?;
261
262    if verbosity != VerbosityLevel::Quiet {
263        println!(
264            "  {} {} sessions from remote, {} from local",
265            "Found".green(),
266            remote_sessions.len(),
267            temp_branch_sessions.len()
268        );
269    }
270
271    // ============================================================================
272    // CONFLICT DETECTION
273    // ============================================================================
274    if verbosity != VerbosityLevel::Quiet {
275        println!("  {} conflicts...", "Detecting".cyan());
276    }
277
278    // Build maps for comparison
279    let remote_map: HashMap<_, _> = remote_sessions
280        .iter()
281        .map(|s| (s.session_id.clone(), s))
282        .collect();
283
284    let local_map: HashMap<_, _> = temp_branch_sessions
285        .iter()
286        .map(|s| (s.session_id.clone(), s))
287        .collect();
288
289    // Find sessions that exist in both and may have conflicts
290    let mut detector = ConflictDetector::new();
291    detector.detect(&temp_branch_sessions, &remote_sessions);
292
293    // ============================================================================
294    // INTERACTIVE CONFIRMATION
295    // ============================================================================
296    if verbosity != VerbosityLevel::Quiet {
297        println!();
298        println!("{}", "Pull Summary:".bold().cyan());
299        println!("  {} Local sessions: {}", "•".cyan(), temp_branch_sessions.len());
300        println!("  {} Remote sessions: {}", "•".cyan(), remote_sessions.len());
301        println!("  {} Conflicts: {}", "•".yellow(), detector.conflict_count());
302        println!();
303    }
304
305    if interactive && interactive_conflict::is_interactive() {
306        let confirm = Confirm::new("Do you want to proceed with merging these changes?")
307            .with_default(true)
308            .with_help_message("This will merge remote sessions with your local changes")
309            .prompt()
310            .context("Failed to get confirmation")?;
311
312        if !confirm {
313            // Clean up temp branch before exiting (force=true to delete even with retention)
314            cleanup_temp_branch(repo.as_ref(), &temp_branch, fetch_remote && state.has_remote, verbosity, 0, true)?;
315            println!("\n{}", "Pull cancelled.".yellow());
316            return Ok(());
317        }
318    }
319
320    // ============================================================================
321    // SMART MERGE AND APPLY TO SYNC REPO
322    // ============================================================================
323    let mut affected_conversations: Vec<ConversationSummary> = Vec::new();
324    let mut merged_count = 0;
325    let mut added_count = 0;
326    let mut modified_count = 0;
327    let mut unchanged_count = 0;
328    let mut skipped_local_newer = 0;
329
330    // Handle conflicts with smart merge
331    if detector.has_conflicts() {
332        if verbosity != VerbosityLevel::Quiet {
333            println!(
334                "  {} {} diverged sessions detected (will create forks)",
335                "!".yellow(),
336                detector.conflict_count()
337            );
338            println!("  {} branches (fork-aware merge)...", "Combining".cyan());
339        }
340
341        let mut smart_merge_success_count = 0;
342        let mut smart_merge_failed_conflicts = Vec::new();
343
344        for conflict in detector.conflicts_mut() {
345            if let (Some(local_session), Some(remote_session)) = (
346                local_map.get(&conflict.session_id),
347                remote_map.get(&conflict.session_id),
348            ) {
349                match conflict.try_smart_merge(local_session, remote_session) {
350                    Ok(()) => {
351                        smart_merge_success_count += 1;
352                        if let crate::conflict::ConflictResolution::SmartMerge {
353                            ref merged_entries,
354                            ref stats,
355                        } = conflict.resolution
356                        {
357                            let merged_session = ConversationSession {
358                                session_id: conflict.session_id.clone(),
359                                entries: merged_entries.clone(),
360                                file_path: conflict.local_file.to_string_lossy().to_string(),
361                            };
362
363                            // Write to sync repo (main branch)
364                            let dest_path = projects_dir.join(
365                                Path::new(&local_session.file_path)
366                                    .strip_prefix(&claude_dir)
367                                    .unwrap_or(Path::new(&local_session.file_path))
368                            );
369                            if let Err(e) = merged_session.write_to_file(&dest_path) {
370                                log::warn!("Failed to write merged session: {}", e);
371                                smart_merge_failed_conflicts.push(conflict.clone());
372                            } else if verbosity != VerbosityLevel::Quiet {
373                                println!(
374                                    "  {} Forked {} ({} local + {} remote = {} combined)",
375                                    "✓".green(),
376                                    conflict.session_id,
377                                    stats.local_messages,
378                                    stats.remote_messages,
379                                    stats.merged_messages,
380                                );
381                            }
382                        }
383                    }
384                    Err(e) => {
385                        log::warn!("Smart merge failed for {}: {}", conflict.session_id, e);
386                        smart_merge_failed_conflicts.push(conflict.clone());
387                    }
388                }
389            }
390        }
391
392        if verbosity != VerbosityLevel::Quiet {
393            println!(
394                "  {} Successfully merged {}/{} diverged sessions",
395                "✓".green(),
396                smart_merge_success_count,
397                detector.conflict_count()
398            );
399        }
400
401        // Handle failed smart merges
402        if !smart_merge_failed_conflicts.is_empty() {
403            if verbosity != VerbosityLevel::Quiet {
404                println!(
405                    "  {} {} conflicts require manual resolution",
406                    "!".yellow(),
407                    smart_merge_failed_conflicts.len()
408                );
409            }
410
411            if crate::interactive_conflict::is_interactive() {
412                let resolution_result = crate::interactive_conflict::resolve_conflicts_interactive(
413                    &mut smart_merge_failed_conflicts,
414                )?;
415
416                let _renames = crate::interactive_conflict::apply_resolutions(
417                    &resolution_result,
418                    &remote_sessions,
419                    &claude_dir,
420                    &projects_dir,
421                )?;
422            } else {
423                // Non-interactive: keep both versions
424                for conflict in &smart_merge_failed_conflicts {
425                    let timestamp = chrono::Utc::now().format("%Y%m%d-%H%M%S");
426                    let conflict_suffix = format!("conflict-{timestamp}");
427
428                    if let Ok(renamed_path) = conflict.clone().resolve_keep_both(&conflict_suffix) {
429                        if let Some(session) = remote_sessions
430                            .iter()
431                            .find(|s| s.session_id == conflict.session_id)
432                        {
433                            session.write_to_file(&renamed_path)?;
434                        }
435                    }
436                }
437            }
438
439            let report = ConflictReport::from_conflicts(detector.conflicts());
440            save_conflict_report(&report)?;
441        }
442    }
443
444    // ============================================================================
445    // MERGE NON-CONFLICTING SESSIONS
446    // ============================================================================
447    if verbosity != VerbosityLevel::Quiet {
448        println!("  {} non-conflicting sessions...", "Merging".cyan());
449    }
450
451    // All sessions from temp branch (local) that aren't conflicts
452    for local_session in &temp_branch_sessions {
453        if detector
454            .conflicts()
455            .iter()
456            .any(|c| c.session_id == local_session.session_id)
457        {
458            continue; // Already handled above
459        }
460
461        let relative_path = Path::new(&local_session.file_path)
462            .strip_prefix(&claude_dir)
463            .ok()
464            .unwrap_or_else(|| Path::new(&local_session.file_path));
465
466        let dest_path = projects_dir.join(relative_path);
467
468        let (operation, should_copy) = if let Some(remote) = remote_map.get(&local_session.session_id) {
469            let relationship = analyze_session_relationship(local_session, remote);
470
471            match relationship {
472                SessionRelationship::Identical => {
473                    unchanged_count += 1;
474                    (SyncOperation::Unchanged, false)
475                }
476                SessionRelationship::LocalIsPrefix => {
477                    // Remote has more - use remote
478                    modified_count += 1;
479                    // Remote is already in main branch, just track it
480                    (SyncOperation::Modified, false)
481                }
482                SessionRelationship::RemoteIsPrefix => {
483                    // Local has more - use local
484                    skipped_local_newer += 1;
485                    (SyncOperation::Modified, true)
486                }
487                SessionRelationship::Diverged => {
488                    // Diverged session not caught by ConflictDetector - do inline merge
489                    // Combine entries from both versions using UUID-based deduplication
490                    // For entries without UUIDs, use (type, timestamp, content_hash) as key
491                    let mut seen_uuids = std::collections::HashSet::new();
492                    let mut seen_non_uuid = std::collections::HashSet::new();
493                    let mut combined_entries = Vec::new();
494
495                    // Helper to create a dedup key for entries without UUIDs
496                    // Uses xxhash for cross-platform stability (same result on ARM and x86)
497                    let make_non_uuid_key = |entry: &crate::parser::ConversationEntry| -> String {
498                        let ts = entry.timestamp.as_deref().unwrap_or("");
499                        let content_hash = entry.message.as_ref()
500                            .map(|m| {
501                                let json = serde_json::to_string(m).unwrap_or_default();
502                                xxhash_rust::xxh3::xxh3_64(json.as_bytes())
503                            })
504                            .unwrap_or(0);
505                        format!("{}:{}:{:016x}", entry.entry_type, ts, content_hash)
506                    };
507
508                    // Add all local entries first
509                    for entry in &local_session.entries {
510                        if let Some(ref uuid) = entry.uuid {
511                            seen_uuids.insert(uuid.clone());
512                        } else {
513                            seen_non_uuid.insert(make_non_uuid_key(entry));
514                        }
515                        combined_entries.push(entry.clone());
516                    }
517
518                    // Add remote entries that aren't already present
519                    for entry in &remote.entries {
520                        let dominated_by_local = if let Some(ref uuid) = entry.uuid {
521                            seen_uuids.contains(uuid)
522                        } else {
523                            seen_non_uuid.contains(&make_non_uuid_key(entry))
524                        };
525                        if !dominated_by_local {
526                            combined_entries.push(entry.clone());
527                        }
528                    }
529
530                    // Sort by timestamp if available
531                    combined_entries.sort_by(|a, b| {
532                        a.timestamp.cmp(&b.timestamp)
533                    });
534
535                    // Write combined session
536                    let merged_session = crate::parser::ConversationSession {
537                        session_id: local_session.session_id.clone(),
538                        entries: combined_entries,
539                        file_path: local_session.file_path.clone(),
540                    };
541                    if let Err(e) = merged_session.write_to_file(&dest_path) {
542                        log::warn!("Failed to write merged diverged session: {}", e);
543                    }
544
545                    modified_count += 1;
546                    (SyncOperation::Modified, false) // Already written above
547                }
548            }
549        } else {
550            // Local-only session
551            added_count += 1;
552            (SyncOperation::Added, true)
553        };
554
555        if should_copy {
556            local_session.write_to_file(&dest_path)?;
557            merged_count += 1;
558        }
559
560        let relative_path_str = relative_path.to_string_lossy().to_string();
561        if let Ok(summary) = ConversationSummary::new(
562            local_session.session_id.clone(),
563            relative_path_str,
564            local_session.latest_timestamp(),
565            local_session.message_count(),
566            operation,
567        ) {
568            affected_conversations.push(summary);
569        }
570    }
571
572    // Also track remote-only sessions (new from remote)
573    for remote_session in &remote_sessions {
574        if local_map.contains_key(&remote_session.session_id) {
575            continue; // Already handled above
576        }
577
578        let relative_path = Path::new(&remote_session.file_path)
579            .strip_prefix(&projects_dir)
580            .ok()
581            .unwrap_or_else(|| Path::new(&remote_session.file_path));
582
583        added_count += 1;
584
585        let relative_path_str = relative_path.to_string_lossy().to_string();
586        if let Ok(summary) = ConversationSummary::new(
587            remote_session.session_id.clone(),
588            relative_path_str,
589            remote_session.latest_timestamp(),
590            remote_session.message_count(),
591            SyncOperation::Added,
592        ) {
593            affected_conversations.push(summary);
594        }
595    }
596
597    // Commit the merged result to main branch
598    repo.stage_all()?;
599    if repo.has_changes()? {
600        let commit_msg = format!(
601            "Merge local changes from {} ({})",
602            temp_branch,
603            chrono::Utc::now().format("%Y-%m-%d %H:%M:%S UTC")
604        );
605        repo.commit(&commit_msg)?;
606    }
607
608    if verbosity != VerbosityLevel::Quiet {
609        println!("  {} Merged {} sessions", "✓".green(), merged_count);
610        if skipped_local_newer > 0 {
611            println!(
612                "  {} Kept {} local sessions (already ahead of remote)",
613                "✓".green(),
614                skipped_local_newer
615            );
616        }
617    }
618
619    // ============================================================================
620    // STEP 6: Append-only merge to .claude
621    // ============================================================================
622    // Key insight: Instead of rewriting files, we APPEND missing entries.
623    // This avoids race conditions with concurrent Claude Code writes.
624    if verbosity != VerbosityLevel::Quiet {
625        println!("  {} to .claude (append-only)...", "Syncing".cyan());
626    }
627
628    // Re-read current local state (may have changed since step 2)
629    let current_local_sessions = discover_sessions(&claude_dir, &filter)?;
630    let current_local_map: HashMap<_, _> = current_local_sessions
631        .iter()
632        .map(|s| (s.session_id.clone(), s))
633        .collect();
634
635    // Read sync repo sessions (contains merged state)
636    let sync_repo_sessions = discover_sessions(&projects_dir, &filter)?;
637
638    let mut sessions_added = 0;
639    let mut sessions_appended = 0;
640    let mut entries_appended = 0;
641
642    for sync_session in &sync_repo_sessions {
643        let relative_path = Path::new(&sync_session.file_path)
644            .strip_prefix(&projects_dir)
645            .unwrap_or(Path::new(&sync_session.file_path));
646        let local_path = claude_dir.join(relative_path);
647
648        if let Some(local_session) = current_local_map.get(&sync_session.session_id) {
649            // Session exists locally - append only missing entries
650
651            // Build sets of what's already in local
652            let local_uuids: HashSet<String> = local_session
653                .entries
654                .iter()
655                .filter_map(|e| e.uuid.clone())
656                .collect();
657
658            let local_non_uuid_keys: HashSet<String> = local_session
659                .entries
660                .iter()
661                .filter(|e| e.uuid.is_none())
662                .map(make_content_key)
663                .collect();
664
665            // Find entries in sync_repo that aren't in local
666            let entries_to_append: Vec<_> = sync_session
667                .entries
668                .iter()
669                .filter(|entry| {
670                    if let Some(ref uuid) = entry.uuid {
671                        !local_uuids.contains(uuid)
672                    } else {
673                        !local_non_uuid_keys.contains(&make_content_key(entry))
674                    }
675                })
676                .cloned()
677                .collect();
678
679            if !entries_to_append.is_empty() {
680                append_entries_to_file(&local_path, &entries_to_append)?;
681                entries_appended += entries_to_append.len();
682                sessions_appended += 1;
683
684                if verbosity == crate::VerbosityLevel::Verbose {
685                    println!(
686                        "    {} +{} entries to {}",
687                        "↳".dimmed(),
688                        entries_to_append.len(),
689                        sync_session.session_id
690                    );
691                }
692            }
693        } else {
694            // Session doesn't exist locally - copy entire file
695            sync_session.write_to_file(&local_path)?;
696            sessions_added += 1;
697
698            if verbosity == crate::VerbosityLevel::Verbose {
699                println!(
700                    "    {} new session {}",
701                    "↳".dimmed(),
702                    sync_session.session_id
703                );
704            }
705        }
706    }
707
708    if verbosity != VerbosityLevel::Quiet {
709        if sessions_added > 0 || sessions_appended > 0 {
710            println!(
711                "  {} Added {} new sessions, appended {} entries to {} sessions",
712                "✓".green(),
713                sessions_added,
714                entries_appended,
715                sessions_appended
716            );
717        } else {
718            println!("  {} No changes needed in .claude", "✓".green());
719        }
720    }
721
722    // ============================================================================
723    // STEP 6b: Merge history.jsonl (session index for --resume picker)
724    // ============================================================================
725    let claude_base_dir = claude_dir.parent().unwrap_or(&claude_dir);
726    let local_history = claude_base_dir.join("history.jsonl");
727    let sync_history = state.sync_repo_path.join("history.jsonl");
728
729    if sync_history.exists() {
730        println!("  {} history.jsonl...", "Merging".cyan());
731        // Merge sync repo entries into local, with local entries taking priority
732        let (total, added) = super::history_merge::merge_history_files(
733            &sync_history,
734            &local_history,
735            super::history_merge::MergePriority::TargetFirst,
736        )?;
737        println!("  {} history.jsonl merged ({} entries, {} new)", "✓".green(), total, added);
738    }
739
740    // ============================================================================
741    // STEP 7: Clean up temp branch (respects retention config)
742    // ============================================================================
743    cleanup_temp_branch(
744        repo.as_ref(),
745        &temp_branch,
746        fetch_remote && state.has_remote,
747        verbosity,
748        filter.temp_branch_retention_hours,
749        false, // don't force delete
750    )?;
751
752    // ============================================================================
753    // CREATE AND SAVE OPERATION RECORD
754    // ============================================================================
755    let operation_record = OperationRecord::new(
756        OperationType::Pull,
757        Some(main_branch.clone()),
758        affected_conversations.clone(),
759    );
760
761    let mut history = match OperationHistory::load() {
762        Ok(h) => h,
763        Err(e) => {
764            log::warn!("Failed to load operation history: {}", e);
765            OperationHistory::default()
766        }
767    };
768
769    if let Err(e) = history.add_operation(operation_record) {
770        log::warn!("Failed to save operation to history: {}", e);
771    }
772
773    // ============================================================================
774    // DISPLAY SUMMARY
775    // ============================================================================
776    if verbosity != VerbosityLevel::Quiet {
777        println!("\n{}", "=== Pull Summary ===".bold().cyan());
778
779        let fork_count = detector.conflict_count();
780        println!(
781            "  {} Added    {} Modified    {} Forked    {} Unchanged",
782            format!("{added_count}").green(),
783            format!("{modified_count}").cyan(),
784            format!("{fork_count}").yellow(),
785            format!("{unchanged_count}").dimmed(),
786        );
787
788        if skipped_local_newer > 0 {
789            println!(
790                "  (Kept {} sessions where local was ahead of remote)",
791                skipped_local_newer
792            );
793        }
794        println!();
795
796        // Group by project
797        let mut by_project: HashMap<String, Vec<&ConversationSummary>> = HashMap::new();
798        for conv in &affected_conversations {
799            if conv.operation == SyncOperation::Unchanged {
800                continue;
801            }
802            let project = conv
803                .project_path
804                .split('/')
805                .next()
806                .unwrap_or("unknown")
807                .to_string();
808            by_project.entry(project).or_default().push(conv);
809        }
810
811        if !by_project.is_empty() {
812            println!("{}", "Affected Conversations:".bold());
813
814            let mut projects: Vec<_> = by_project.keys().collect();
815            projects.sort();
816
817            for project in projects {
818                let conversations = &by_project[project];
819                println!("\n  {} {}/", "Project:".bold(), project.cyan());
820
821                for conv in conversations.iter().take(MAX_CONVERSATIONS_TO_DISPLAY) {
822                    let operation_str = match conv.operation {
823                        SyncOperation::Added => "ADD".green(),
824                        SyncOperation::Modified => "MOD".cyan(),
825                        SyncOperation::Conflict => "FORK".yellow(),
826                        SyncOperation::Unchanged => "---".dimmed(),
827                    };
828
829                    let timestamp_str = conv
830                        .timestamp
831                        .as_ref()
832                        .and_then(|t| t.split('T').next())
833                        .unwrap_or("unknown");
834
835                    println!(
836                        "    {} {} ({}msg, {})",
837                        operation_str,
838                        conv.project_path,
839                        conv.message_count,
840                        timestamp_str.dimmed()
841                    );
842                }
843
844                if conversations.len() > MAX_CONVERSATIONS_TO_DISPLAY {
845                    println!(
846                        "    {} ... and {} more conversations",
847                        "...".dimmed(),
848                        conversations.len() - MAX_CONVERSATIONS_TO_DISPLAY
849                    );
850                }
851            }
852        }
853
854        println!("\n{}", "Pull complete!".green().bold());
855    }
856
857    Ok(())
858}

860/// Clean up the temporary branch (local and optionally remote)
861///
862/// If retention_hours > 0, skip deletion (branch will be cleaned up later).
863/// If force is true, always delete (used when pull is cancelled).
864fn cleanup_temp_branch(
865    repo: &dyn scm::Scm,
866    temp_branch: &str,
867    has_remote: bool,
868    verbosity: crate::VerbosityLevel,
869    retention_hours: u32,
870    force: bool,
871) -> Result<()> {
872    use crate::VerbosityLevel;
873
874    // Skip cleanup if retention is enabled and this isn't a forced cleanup
875    if retention_hours > 0 && !force {
876        if verbosity != VerbosityLevel::Quiet {
877            println!(
878                "  {} Temp branch {} retained for {} hours",
879                "ℹ".cyan(),
880                temp_branch,
881                retention_hours
882            );
883        }
884        return Ok(());
885    }
886
887    if verbosity != VerbosityLevel::Quiet {
888        println!("  {} temp branch...", "Cleaning up".cyan());
889    }
890
891    // Delete remote branch first (if it exists)
892    if has_remote {
893        match repo.delete_remote_branch("origin", temp_branch) {
894            Ok(_) => {
895                if verbosity != VerbosityLevel::Quiet {
896                    println!("  {} Deleted origin/{}", "✓".green(), temp_branch);
897                }
898            }
899            Err(e) => {
900                log::debug!("Failed to delete remote branch (may not exist): {}", e);
901            }
902        }
903    }
904
905    // Delete local branch
906    match repo.delete_branch(temp_branch) {
907        Ok(_) => {
908            if verbosity != VerbosityLevel::Quiet {
909                println!("  {} Deleted local branch {}", "✓".green(), temp_branch);
910            }
911        }
912        Err(e) => {
913            log::warn!("Failed to delete local branch: {}", e);
914        }
915    }
916
917    Ok(())
918}

920/// Clean up old temporary branches that have exceeded their retention period
921fn cleanup_old_temp_branches(
922    repo: &dyn scm::Scm,
923    has_remote: bool,
924    retention_hours: u32,
925    verbosity: crate::VerbosityLevel,
926) -> Result<()> {
927    use crate::VerbosityLevel;
928
929    // If retention is 0, branches are deleted immediately so nothing to clean up
930    if retention_hours == 0 {
931        return Ok(());
932    }
933
934    // Get list of local branches matching our temp branch pattern
935    let branches = match repo.list_branches() {
936        Ok(b) => b,
937        Err(e) => {
938            log::debug!("Failed to list branches for cleanup: {}", e);
939            return Ok(());
940        }
941    };
942
943    let now = chrono::Utc::now();
944    let retention_duration = chrono::Duration::hours(retention_hours as i64);
945    let mut cleaned = 0;
946
947    for branch in branches {
948        // Only process our temp branches (format: sync-local-YYYYMMDD-HHMMSS)
949        if !branch.starts_with("sync-local-") {
950            continue;
951        }
952
953        // Parse timestamp from branch name
954        let timestamp_part = branch.strip_prefix("sync-local-").unwrap_or(&branch);
955        if let Ok(branch_time) = chrono::NaiveDateTime::parse_from_str(timestamp_part, "%Y%m%d-%H%M%S")
956        {
957            let branch_time_utc = chrono::DateTime::<chrono::Utc>::from_naive_utc_and_offset(
958                branch_time,
959                chrono::Utc,
960            );
961
962            // Check if branch has exceeded retention period
963            if now.signed_duration_since(branch_time_utc) > retention_duration {
964                log::debug!("Cleaning up old temp branch: {}", branch);
965
966                // Delete remote branch first
967                if has_remote {
968                    if let Err(e) = repo.delete_remote_branch("origin", &branch) {
969                        log::debug!("Failed to delete remote branch {}: {}", branch, e);
970                    }
971                }
972
973                // Delete local branch
974                if let Err(e) = repo.delete_branch(&branch) {
975                    log::debug!("Failed to delete local branch {}: {}", branch, e);
976                } else {
977                    cleaned += 1;
978                }
979            }
980        }
981    }
982
983    if cleaned > 0 && verbosity != VerbosityLevel::Quiet {
984        println!(
985            "  {} Cleaned up {} old temp branch{}",
986            "✓".green(),
987            cleaned,
988            if cleaned == 1 { "" } else { "es" }
989        );
990    }
991
992    Ok(())
993}
