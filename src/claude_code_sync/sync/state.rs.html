5/// Sync state and configuration
6///
7/// This struct stores the persistent state of the Claude Code sync system.
8/// It tracks where the sync repository is located, whether it's configured
9/// with a remote, and whether it was originally cloned from a remote URL.
10///
11/// The state is serialized to JSON and stored in the user's configuration
12/// directory, allowing the sync system to remember its configuration across
13/// multiple command invocations.
14#[derive(Debug, serde::Serialize, serde::Deserialize)]
15pub struct SyncState {
16    /// Path to the local git repository used for syncing Claude Code conversations
17    ///
18    /// This is the directory where all conversation sessions are stored in git format.
19    /// The conversations are organized under a `projects/` subdirectory within this path.
20    /// This repository can be a local-only git repository or one that's synchronized
21    /// with a remote origin.
22    pub sync_repo_path: PathBuf,
23
24    /// Whether the sync repository has a remote configured
25    ///
26    /// When `true`, the repository has a remote (typically named "origin") configured,
27    /// allowing push and pull operations to synchronize with a remote git service
28    /// (e.g., GitHub, GitLab). When `false`, the repository is local-only and cannot
29    /// push to or pull from remote servers.
30    pub has_remote: bool,
31
32    /// Whether the repository was cloned from a remote URL
33    ///
34    /// This field distinguishes between repositories that were:
35    /// - Cloned from an existing remote repository (`true`)
36    /// - Initialized locally and optionally had a remote added later (`false`)
37    ///
38    /// This affects certain initialization behaviors, as cloned repositories
39    /// may already have existing content and history.
40    #[serde(default)]
41    pub is_cloned_repo: bool,
42}
43
44impl SyncState {
45    /// Loads the sync state from the user's configuration directory
46    ///
47    /// This function reads the persisted sync configuration from disk and deserializes
48    /// it into a `SyncState` instance. The state file is stored in the user's Claude
49    /// configuration directory and contains information about the sync repository location,
50    /// remote configuration, and initialization method.
51    ///
52    /// # Returns
53    ///
54    /// Returns a `Result` containing the loaded `SyncState` on success.
55    ///
56    /// # Errors
57    ///
58    /// This function returns an error if:
59    /// - The sync system has not been initialized (state file doesn't exist)
60    /// - The state file cannot be read (permission errors, I/O errors)
61    /// - The state file contains invalid JSON or cannot be deserialized
62    ///
63    /// If the sync is not initialized, the error message will instruct the user
64    /// to run `claude-code-sync init` first.
65    ///
66    /// # Examples
67    ///
68    /// ```
69    /// # use claude_code_sync::sync::SyncState;
70    /// # fn example() -> Result<(), Box<dyn std::error::Error>> {
71    /// match SyncState::load() {
72    ///     Ok(state) => {
73    ///         println!("Sync repo: {}", state.sync_repo_path.display());
74    ///         println!("Has remote: {}", state.has_remote);
75    ///     }
76    ///     Err(e) => log::error!("Failed to load sync state: {}", e),
77    /// }
78    /// # Ok(())
79    /// # }
80    /// ```
81    pub fn load() -> Result<Self> {
82        let state_path = Self::state_file_path()?;
83
84        if !state_path.exists() {
85            return Err(anyhow!(
86                "Sync not initialized. Run 'claude-code-sync init' first."
87            ));
88        }
89
90        let content = fs::read_to_string(&state_path).context("Failed to read sync state")?;
91
92        let state: SyncState =
93            serde_json::from_str(&content).context("Failed to parse sync state")?;
94
95        Ok(state)
96    }
97
98    pub(crate) fn save(&self) -> Result<()> {
99        let state_path = Self::state_file_path()?;
100
101        if let Some(parent) = state_path.parent() {
102            fs::create_dir_all(parent)?;
103        }
104
105        let content =
106            serde_json::to_string_pretty(self).context("Failed to serialize sync state")?;
107
108        fs::write(&state_path, content).context("Failed to write sync state")?;
109
110        Ok(())
111    }
112
113    fn state_file_path() -> Result<PathBuf> {
114        crate::config::ConfigManager::state_file_path()
115    }
116}
