1//! Verify session sync between two .claude directories
2//!
3//! Compares session files to ensure they're identical or one is a prefix of the other
4//! (same entries, just one has more recent messages appended).
5
6use anyhow::{Context, Result};
7use std::collections::{HashMap, HashSet};
8use std::fs;
9use std::path::{Path, PathBuf};
10use walkdir::WalkDir;
11
12#[derive(Debug, Clone)]
13struct SessionInfo {
14    path: PathBuf,
15    relative_path: String,
16    entry_count: usize,
17    /// UUIDs in order - used to detect prefix relationships
18    uuids: Vec<String>,
19}
20
21#[derive(Debug, Default)]
22struct ComparisonStats {
23    identical: usize,
24    host1_ahead: usize,
25    host2_ahead: usize,
26    diverged: usize,
27    host1_only: usize,
28    host2_only: usize,
29}
30
31fn discover_sessions(base_path: &Path) -> Result<HashMap<String, SessionInfo>> {
32    let mut sessions = HashMap::new();
33
34    for entry in WalkDir::new(base_path)
35        .follow_links(false)
36        .into_iter()
37        .filter_map(|e| e.ok())
38    {
39        let path = entry.path();
40        if path.extension().and_then(|s| s.to_str()) != Some("jsonl") {
41            continue;
42        }
43
44        let relative_path = path
45            .strip_prefix(base_path)
46            .unwrap_or(path)
47            .to_string_lossy()
48            .to_string();
49
50        match parse_session_uuids(path) {
51            Ok((entry_count, uuids)) => {
52                sessions.insert(
53                    relative_path.clone(),
54                    SessionInfo {
55                        path: path.to_path_buf(),
56                        relative_path,
57                        entry_count,
58                        uuids,
59                    },
60                );
61            }
62            Err(e) => {
63                eprintln!("Warning: Failed to parse {}: {}", path.display(), e);
64            }
65        }
66    }
67
68    Ok(sessions)
69}
70
71fn parse_session_uuids(path: &Path) -> Result<(usize, Vec<String>)> {
72    let content = fs::read_to_string(path).context("Failed to read file")?;
73    let mut uuids = Vec::new();
74    let mut entry_count = 0;
75
76    for line in content.lines() {
77        if line.trim().is_empty() {
78            continue;
79        }
80        entry_count += 1;
81
82        // Parse JSON and extract uuid
83        if let Ok(value) = serde_json::from_str::<serde_json::Value>(line) {
84            if let Some(uuid) = value.get("uuid").and_then(|v| v.as_str()) {
85                uuids.push(uuid.to_string());
86            }
87        }
88    }
89
90    Ok((entry_count, uuids))
91}

93/// Check if vec1 is a prefix of vec2
94fn is_prefix(shorter: &[String], longer: &[String]) -> bool {
95    if shorter.len() > longer.len() {
96        return false;
97    }
98    shorter.iter().zip(longer.iter()).all(|(a, b)| a == b)
99}

101fn compare_sessions(
102    host1_sessions: &HashMap<String, SessionInfo>,
103    host2_sessions: &HashMap<String, SessionInfo>,
104    host1_name: &str,
105    host2_name: &str,
106) -> (ComparisonStats, Vec<(String, SessionInfo, SessionInfo)>) {
107    let mut stats = ComparisonStats::default();
108    let mut diverged_sessions = Vec::new();
109
110    // Get all unique paths
111    let all_paths: HashSet<_> = host1_sessions
112        .keys()
113        .chain(host2_sessions.keys())
114        .collect();
115
116    for path in all_paths {
117        let host1_info = host1_sessions.get(path);
118        let host2_info = host2_sessions.get(path);
119
120        match (host1_info, host2_info) {
121            (None, Some(_)) => {
122                stats.host2_only += 1;
123            }
124            (Some(_), None) => {
125                stats.host1_only += 1;
126            }
127            (Some(h1), Some(h2)) => {
128                if h1.uuids == h2.uuids {
129                    stats.identical += 1;
130                } else if is_prefix(&h1.uuids, &h2.uuids) {
131                    // host1 is prefix of host2 - host2 is ahead
132                    stats.host2_ahead += 1;
133                } else if is_prefix(&h2.uuids, &h1.uuids) {
134                    // host2 is prefix of host1 - host1 is ahead
135                    stats.host1_ahead += 1;
136                } else {
137                    // Diverged
138                    stats.diverged += 1;
139                    diverged_sessions.push((path.clone(), h1.clone(), h2.clone()));
140                }
141            }
142            (None, None) => unreachable!(),
143        }
144    }
145
146    (stats, diverged_sessions)
147}

149fn find_divergence_point(uuids1: &[String], uuids2: &[String]) -> usize {
150    uuids1
151        .iter()
152        .zip(uuids2.iter())
153        .position(|(a, b)| a != b)
154        .unwrap_or(uuids1.len().min(uuids2.len()))
155}

157fn main() -> Result<()> {
158    let args: Vec<String> = std::env::args().collect();
159
160    if args.len() != 3 {
161        eprintln!("Usage: verify-sync <path1> <path2>");
162        eprintln!();
163        eprintln!("Compares two .claude/projects directories to verify sync status.");
164        eprintln!("Sessions should be identical or one should be a prefix of the other.");
165        eprintln!();
166        eprintln!("Example:");
167        eprintln!("  verify-sync /tmp/arm-claude /tmp/x86-claude");
168        std::process::exit(1);
169    }
170
171    let path1 = PathBuf::from(&args[1]);
172    let path2 = PathBuf::from(&args[2]);
173
174    // Extract host names from paths for display
175    let host1_name = path1
176        .file_name()
177        .map(|s| s.to_string_lossy().to_string())
178        .unwrap_or_else(|| "host1".to_string());
179    let host2_name = path2
180        .file_name()
181        .map(|s| s.to_string_lossy().to_string())
182        .unwrap_or_else(|| "host2".to_string());
183
184    println!("=== Claude Code Session Sync Verification ===");
185    println!();
186
187    println!("Scanning {}...", path1.display());
188    let host1_sessions = discover_sessions(&path1)?;
189    println!("  Found {} sessions", host1_sessions.len());
190
191    println!("Scanning {}...", path2.display());
192    let host2_sessions = discover_sessions(&path2)?;
193    println!("  Found {} sessions", host2_sessions.len());
194
195    println!();
196    println!("=== Comparing Sessions ===");
197
198    let (stats, diverged) =
199        compare_sessions(&host1_sessions, &host2_sessions, &host1_name, &host2_name);
200
201    println!();
202    println!("Results:");
203    println!("  ✓ Identical:      {}", stats.identical);
204    println!("  → {} ahead:  {}", host1_name, stats.host1_ahead);
205    println!("  ← {} ahead:  {}", host2_name, stats.host2_ahead);
206    println!("  ✗ Diverged:       {}", stats.diverged);
207    println!("  ◦ {} only:   {}", host1_name, stats.host1_only);
208    println!("  ◦ {} only:   {}", host2_name, stats.host2_only);
209    println!();
210
211    let total_shared = stats.identical + stats.host1_ahead + stats.host2_ahead + stats.diverged;
212
213    if stats.diverged == 0 {
214        println!(
215            "✅ All {} shared sessions are in sync (one is prefix of other)",
216            total_shared
217        );
218    } else {
219        println!(
220            "⚠️  {} sessions have diverged histories!",
221            stats.diverged
222        );
223        println!();
224        println!("=== Diverged Session Details ===");
225
226        for (path, h1, h2) in diverged.iter().take(10) {
227            let diverge_point = find_divergence_point(&h1.uuids, &h2.uuids);
228
229            println!();
230            println!("Session: {}", path);
231            println!(
232                "  {} entries: {}, {} entries: {}",
233                host1_name, h1.entry_count, host2_name, h2.entry_count
234            );
235            println!(
236                "  Divergence at entry {} (0-indexed)",
237                diverge_point
238            );
239
240            if diverge_point < h1.uuids.len() && diverge_point < h2.uuids.len() {
241                println!(
242                    "  {} UUID at divergence: {}",
243                    host1_name,
244                    &h1.uuids[diverge_point]
245                );
246                println!(
247                    "  {} UUID at divergence: {}",
248                    host2_name,
249                    &h2.uuids[diverge_point]
250                );
251            }
252
253            // Show context: entries before divergence
254            if diverge_point > 0 {
255                println!(
256                    "  Last common UUID: {}",
257                    &h1.uuids[diverge_point - 1]
258                );
259            }
260        }
261
262        if diverged.len() > 10 {
263            println!();
264            println!("... and {} more diverged sessions", diverged.len() - 10);
265        }
266    }
267
268    // Exit with error code if there are diverged sessions
269    if stats.diverged > 0 {
270        std::process::exit(1);
271    }
272
273    Ok(())
274}
